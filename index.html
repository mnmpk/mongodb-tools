SCRIPT=$(cat << 'ENDSCRIPT'
#!/bin/bash
### BEGIN INIT INFO
# Provides:          disable-transparent-hugepages
# Required-Start:    $local_fs
# Required-Stop:
# X-Start-Before:    mongod mongodb-mms-automation-agent
# Default-Start:     2 3 4 5
# Default-Stop:      0 1 6
# Short-Description: Disable Linux transparent huge pages
# Description:       Disable Linux transparent huge pages, to improve
#                    database performance.
### END INIT INFO

case $1 in
    start)
    if [ -d /sys/kernel/mm/transparent_hugepage ]; then
        thp_path=/sys/kernel/mm/transparent_hugepage
    elif [ -d /sys/kernel/mm/redhat_transparent_hugepage ]; then
        thp_path=/sys/kernel/mm/redhat_transparent_hugepage
    else
        return 0
    fi

    echo 'never' > ${thp_path}/enabled
    echo 'never' > ${thp_path}/defrag

    re='^[0-1]+$'
    if [[ $(cat ${thp_path}/khugepaged/defrag) =~ $re ]]
    then
        # RHEL 7
        echo 0  > ${thp_path}/khugepaged/defrag
    else
        # RHEL 6
        echo 'no' > ${thp_path}/khugepaged/defrag
    fi

    #Set Readahead for Data Disk
    blockdev --setra 8 /dev/xvdb
    unset re
    unset thp_path
    ;;
esac
ENDSCRIPT
)

echo "$SCRIPT" | sudo tee /etc/init.d/disable-transparent-hugepages

sudo chmod 755 /etc/init.d/disable-transparent-hugepages

sudo chkconfig --add disable-transparent-hugepages


sudo reboot
ENDPRODNOTES

# Allow copying of extensions from CSRs to the resulting certificates
sudo sed -i 's/^# copy_extensions = copy.*/copy_extensions = copy/' /etc/pki/tls/openssl.cnf

# Setup the tracking files for certificate registration
sudo /bin/rm -f /etc/pki/CA/index*
sudo /bin/rm -f /etc/pki/CA/serial*

sudo touch /etc/pki/CA/index.txt
echo 1000 | sudo tee  /etc/pki/CA/serial

host0="om-1-witty-mayfly.mdbtraining.net"
host1="om-2-witty-mayfly.mdbtraining.net"
host2="om-3-witty-mayfly.mdbtraining.net"
loadbalancer="witty-mayfly-om.mdbtraining.net"
clusterdesc="OpsManager"

# Generate the key files and CSRs for each host, include the SAN for the load balancer and the host itself
openssl11 req -newkey rsa:2048 -nodes -new -addext "subjectAltName=DNS:${loadbalancer},DNS:${host0}" -sha256 -out ${host0}.csr -keyout ${host0}.key -subj "/C=GB/ST=Scotland/L=Glasgow/O=MongoDB/OU=${clusterdesc}/CN=${host0}"
openssl11 req -newkey rsa:2048 -nodes -new -addext "subjectAltName=DNS:${loadbalancer},DNS:${host1}" -sha256 -out ${host1}.csr -keyout ${host1}.key -subj "/C=GB/ST=Scotland/L=Glasgow/O=MongoDB/OU=${clusterdesc}/CN=${host1}"
openssl11 req -newkey rsa:2048 -nodes -new -addext "subjectAltName=DNS:${loadbalancer},DNS:${host2}" -sha256 -out ${host2}.csr -keyout ${host2}.key -subj "/C=GB/ST=Scotland/L=Glasgow/O=MongoDB/OU=${clusterdesc}/CN=${host2}"

# Sign the certificates for each host
sudo openssl11 ca -notext -in ${host0}.csr -out ${host0}.cert -keyfile ca.key -cert ca.cert -outdir . -batch
sudo openssl11 ca -notext -in ${host1}.csr -out ${host1}.cert -keyfile ca.key -cert ca.cert -outdir . -batch
sudo openssl11 ca -notext -in ${host2}.csr -out ${host2}.cert -keyfile ca.key -cert ca.cert -outdir . -batch

# concatenate the private key and the certificate into a PEM file
cat ${host0}.key ${host0}.cert > ${host0}.pem
cat ${host1}.key ${host1}.cert > ${host1}.pem
cat ${host2}.key ${host2}.cert > ${host2}.pem
